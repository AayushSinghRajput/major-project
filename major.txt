# 1. Check your current branch
git branch

# 2. Stash or commit any uncommitted changes
git status
git add .
git commit -m "WIP: Save changes before merge"
# or use
git stash

# 3. Switch to the main branch
git checkout main

# 4. Pull the latest changes from the remote main
git pull origin main

# 5. Merge your feature branch into main
git merge feature-branch-name

# 6. (Optional) Resolve any merge conflicts if prompted
# -- Open the conflicted files, fix them manually, then:
git add .
git commit -m "Resolved merge conflicts"

# 7. Push the updated main branch to remote
git push origin main

# 8. (Optional) Delete the feature branch locally and remotely
git branch -d feature-branch-name        # delete local
git push origin --delete feature-branch-name  # delete remote
